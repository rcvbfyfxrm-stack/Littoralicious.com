```bash
cd /tmp/twa; DATE=$(date +%Y-%m-%d)
[ -s theworldaffair/$DATE.html ] && [ -s theworldaffair/$DATE/synthese.html ] || { echo "ERR: pages missing"; exit 1; }

import subprocess, json, base64, sys, glob
DATE=sys.argv[1]; REPO="rcvbfyfxrm-stack/Littoralicious.com"; BR="rebuild/publishing-system"
def gh(args, body=None):
    p=subprocess.run(["gh","api"]+args, input=body, capture_output=True, text=True)
    if p.returncode!=0: print("gh api FAILED:", " ".join(args), p.stderr[:400]); sys.exit(1)

files=[f"theworldaffair/{DATE}.html","theworldaffair/editions.json"]+sorted(glob.glob(f"theworldaffair/{DATE}/*.html"))
ref=gh([f"repos/{REPO}/git/ref/heads/{BR}"]); base=ref["object"]["sha"]
basetree=gh([f"repos/{REPO}/git/commits/{base}"])["tree"]["sha"]
tree=[]
for f in files:
    b64=base64.b64encode(open(f,"rb").read()).decode()
    blob=gh([f"repos/{REPO}/git/blobs","-X","POST","--input","-"], json.dumps({"content":b64,"encoding":"base64"}))

gh([f"repos/{REPO}/git/refs/heads/{BR}","-X","PATCH","--input","-"], json.dumps({"sha":c["sha"]}))
print("PUBLISHED", c["sha"])
PYEOF
```
